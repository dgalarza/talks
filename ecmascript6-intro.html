Scope

Share same lexical this as surrounding code

                
                  var page = {
                    init: function() {
                      this.bindEvents();
                    },

                    bindEvents: function() {
                      $('body').on('ajax:success', (_, response) => {
                        this.handleResponse(response);
                      });
                    },

                    handleResponse: function(response) { }
                  };

                  page.init();

Arrow functions as expressions

                
                  var people = [
                    {
                      name: "Ralph"
                    },
                    {
                      name: "Damian"
                    }
                  ];

                  // ["Ralph", "Damian"]
                  var names = people.map(person => person.name);

Classes

Inheritance

                
                  class PokerHand {
                    constructor(cards) {
                      this.cards = cards;
                    }
                  };

                  class OnePair extends PokerHand {
                    constructor(cards) {
                      super(cards);
                    }
                  };

String interpolation

                
                  var name = "Ralph";

                  console.log(`Hello ${name}`);

Destructuring

From an array

                
                  var [x, y] = [20, 30, 0];

                  console.log(x);
                  console.log(y);

From an object

                
                  var person = {
                    name: "Ralph",
                    address: "1384 Broadway"
                  };

                  var { name, address } = person;

                  console.log(name);
                  console.log(address);

Function arguments

                
                  var person = {
                    name: "Ralph",
                    address: "1384 Broadway"
                  };

                  function greet({ name }) {
                    console.log(`Hello ${name}`);
                  }

                  // Hello Ralph
                  greet(person);
